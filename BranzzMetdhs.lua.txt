		end)

-- FunÃ§Ã£o para animar botÃ£o ON/OFF
local function toggleButton(button, state)
	if state then
		-- Animar para verde
		local tween = TweenService:Create(button, TweenInfo.new(0.3), {
			BackgroundColor3 = Color3.fromRGB(0, 150, 0)
		})
		tween:Play()
		button.Text = button.Text:gsub("%(OFF%)", "(ON)"):gsub("%(ON%)", "(ON)")
	else
		-- Animar para vermelho
		local tween = TweenService:Create(button, TweenInfo.new(0.3), {
			BackgroundColor3 = Color3.fromRGB(150, 0, 0)
		})
		tween:Play()
		button.Text = button.Text:gsub("%(ON%)", "(OFF)"):gsub("%(OFF%)", "(OFF)")
	end
end

-- 2. Sistema de Voo Livre
local function setupFlight()
	if flightConnection then
		flightConnection:Disconnect()
		flightConnection = nil
	end
	
	local flightSpeed = 28
	local flying = false
	
	FlightButton.MouseButton1Click:Connect(function()
		flying = not flying
		flightEnabled = flying
		
		toggleButton(FlightButton, flying)
		
		if flying then
			FlightButton.Text = "VÃ´o Livre ðŸ¦… (ON)"
			
			-- Sistema de voo
			flightConnection = RunService.RenderStepped:Connect(function()
				if flying and player.Character then
					local humanoid = player.Character:FindFirstChild("Humanoid")
					local rootPart = player.Character:FindFirstChild("HumanoidRootPart")
					
					if humanoid and rootPart then
						-- Ativar voo no humanoid
						humanoid.PlatformStand = true
						
						-- Mover na direÃ§Ã£o da cÃ¢mera
						local camera = workspace.CurrentCamera
						if camera then
							local direction = camera.CFrame.LookVector
							rootPart.Velocity = direction * flightSpeed
						end
					end
				end
			end)
		else
			FlightButton.Text = "VÃ´o Livre ðŸ¦… (OFF)"
			
			-- Resetar personagem
			if player.Character then
				local humanoid = player.Character:FindFirstChild("Humanoid")
				if humanoid then
					humanoid.PlatformStand = false
				end
			end
		end
	end)
end

setupFlight()

-- 3. FunÃ§Ã£o TP Position
local function createPlatform(position)
	local platform = Instance.new("Part")
	platform.Size = Vector3.new(10, 1, 10)
	platform.Position = position
	platform.Anchored = true
	platform.CanCollide = true
	platform.Transparency = 0.5
	platform.Material = EnumMaterial.Neon
	platform.BrickColor = BrickColor.new("Bright violet")
	platform.Parent = workspace
	
	table.insert(platformParts, platform)
	
	-- Remover apÃ³s 8 segundos
	wait(8)
	if platform and platform.Parent then
		platform:Destroy()
	end
	
	-- Remover da tabela
	for i, p in ipairs(platformParts) do
		if p == platform then
			table.remove(platformParts, i)
			break
		end
	end
end

TPButton.MouseButton1Click:Connect(function()
	if not savedPosition then
		TPButton.Text = "Marque uma posiÃ§Ã£o primeiro!"
		wait(1)
		TPButton.Text = "TP POSITION ðŸ“ ðŸŒ"
		return
	end
	
	if not player.Character then return end
	
	local rootPart = player.Character:FindFirstChild("HumanoidRootPart")
	if not rootPart then return end
	
	local startPos = rootPart.Position
	local targetPos = savedPosition
	local direction = (targetPos - startPos).Unit
	
	-- Ativar voo temporariamente
	flightEnabled = true
	local originalFlight = flightEnabled
	
	-- Configurar voo para TP
	local tpFlightSpeed = 50
	local distance = (targetPos - startPos).Magnitude
	local steps = math.floor(distance / 10)
	
	-- Criar plataformas no caminho
	coroutine.wrap(function()
		for i = 1, steps do
			local platformPos = startPos + (direction * (i * 10))
			createPlatform(platformPos)
			wait(3)
		end
	end)()
	
	-- Mover para a posiÃ§Ã£o
	coroutine.wrap(function()
		local startTime = tick()
		while (rootPart.Position - targetPos).Magnitude > 5 and tick() - startTime < 10 do
			rootPart.Velocity = direction * tpFlightSpeed
			wait()
		end
		
		-- Chegou no destino
		rootPart.Velocity = Vector3.new(0, 0, 0)
		rootPart.Position = targetPos
		flightEnabled = originalFlight
	end)()
end)

DesyncV2Button.MouseButton1Click:Connect(function()
	desyncV2Enabled = not desyncV2Enabled
	toggleButton(DesyncV2Button, desyncV2Enabled)
	
	if desyncV2Enabled then
		DesyncV2Button.Text = "Desync v2 (ON)"
		
		-- Sistema melhorado de lag
		coroutine.wrap(function()
			while desyncV2Enabled and player.Character do
				-- Movimento aleatÃ³rio para simular lag
				local rootPart = player.Character:FindFirstChild("HumanoidRootPart")
				if rootPart then
					-- Aplicar pequenos movimentos aleatÃ³rios
					local randomOffset = Vector3.new(
						math.random(-5, 5),
						math.random(-2, 2),
						math.random(-5, 5)
					)
					
					rootPart.CFrame = rootPart.CFrame + randomOffset
					
					-- Delay variÃ¡vel para simular lag
					wait(math.random(0.1, 0.3))
				else
					wait(0.5)
				end
			end
		end)()
	else
		DesyncV2Button.Text = "Desync v2 (OFF)"
	end
end)

HitboxButton.MouseButton1Click:Connect(function()

	end
	
	if hitboxEnabled then
		HitboxButton.Text = "ON"
		
		-- Criar hitbox expandida
		local size = tonumber(SizeBox.Text) or 5
		
		if player.Character then
			local rootPart = player.Character:FindFirstChild("HumanoidRootPart")
			if rootPart then
				currentHitbox = Instance.new("Part")
				currentHitbox.Name = "ExpandedHitbox"
				currentHitbox.Size = Vector3.new(size, size, size)
				currentHitbox.Position = rootPart.Position
				currentHitbox.Anchored = true
				currentHitbox.CanCollide = true
				currentHitbox.Transparency = 0.7
				currentHitbox.Material = EnumMaterial.Neon
				currentHitbox.BrickColor = BrickColor.new("White")
				currentHitbox.Parent = workspace
				
				-- Seguir o jogador
				coroutine.wrap(function()
					while hitboxEnabled and currentHitbox do
						if rootPart and rootPart.Parent then
							currentHitbox.Position = rootPart.Position
						end
						wait()
					end
				end)()
			end
		end
	else
		HitboxButton.Text = "OFF"
	end
end)
