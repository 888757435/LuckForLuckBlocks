local function toggleFlight()
    flying = not flying
    
    if flying then
        FlightButton.Text = "ü¶Ö V√¥o Livre: ON"
        FlightIndicator.BackgroundColor3 = Color3.fromRGB(50, 180, 50)
        
        if flightConnection then
            flightConnection:Disconnect()
        end
        
        flightConnection = RunService.RenderStepped:Connect(function()
            if flying and player.Character then
                local camera = workspace.CurrentCamera
                local rootPart = player.Character:FindFirstChild("HumanoidRootPart")
                
                if camera and rootPart then
                    rootPart.Velocity = camera.CFrame.LookVector * flightSpeed
                end
            end
        end)
    else
        FlightButton.Text = "ü¶Ö V√¥o Livre: OFF"
        FlightIndicator.BackgroundColor3 = Color3.fromRGB(180, 50, 50)
        
        if flightConnection then
            flightConnection:Disconnect()
            flightConnection = nil
        end
        
        if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
            player.Character.HumanoidRootPart.Velocity = Vector3.new(0, 0, 0)
        end
    end
end

-- Fun√ß√£o criar plataforma
local function createPlatform(position)
    local platform = Instance.new("Part")
    platform.Name = "TempPlatform"
    platform.Size = Vector3.new(8, 1, 8)
    platform.Position = position
    platform.Anchored = true
    platform.CanCollide = true
    platform.Transparency = 0.3
    platform.Color = Color3.fromRGB(255, 255, 255)
    platform.Material = EnumMaterial.Neon
    platform.Parent = workspace
    
    table.insert(platforms, platform)
    
    task.delay(8, function()
        if platform and platform.Parent then
            platform:Destroy()
        end
    end)
end

-- Fun√ß√£o TP com plataformas
local function teleportWithPlatforms()
    if not savedPosition then
        MarkStatus.Text = "Salve uma posi√ß√£o primeiro!"
        MarkStatus.TextColor3 = Color3.fromRGB(255, 100, 100)
        task.wait(2)
        MarkStatus.TextColor3 = Color3.fromRGB(200, 200, 200)
        return
    end
    
    if not player.Character or not player.Character:FindFirstChild("HumanoidRootPart") then
        return
    end
    
    local root = player.Character.HumanoidRootPart
    local startPos = root.Position
    local direction = (savedPosition - startPos).Unit
    local distance = (savedPosition - startPos).Magnitude
    
    -- Ativar voo
    local wasFlying = flying
    if not flying then
        toggleFlight()
    end
    
    -- Criar plataformas
    local steps = math.ceil(distance / 20)
    for i = 1, steps do
        task.spawn(function()
            task.wait(i * 0.2)
            local platformPos = startPos + (direction * (i * 20))
            createPlatform(platformPos + Vector3.new(0, -5, 0))
        end)
    end
    
    -- Mover at√© a posi√ß√£o
    local travelTime = distance / flightSpeed
    local startTime = tick()
    
    while tick() - startTime < travelTime do
        if not player.Character or not player.Character:FindFirstChild("HumanoidRootPart") then
            break
        end
        RunService.Heartbeat:Wait()
    end
    
    -- Teleportar final
    if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
        player.Character.HumanoidRootPart.CFrame = CFrame.new(savedPosition)
    end
    
    -- Criar plataforma final
    createPlatform(savedPosition + Vector3.new(0, -5, 0))
    
    -- Desativar voo se n√£o estava ativo antes
    if not wasFlying then
        task.wait(0.5)
        toggleFlight()
    end
end

DesyncV1Indicator.BackgroundColor3 = Color3.fromRGB(180, 50, 50)
local desync1Corner = Instance.new("UICorner")
desync1Corner.CornerRadius = UDim.new(1, 0)
desync1Corner.Parent = DesyncV1Indicator
DesyncV1Indicator.Parent = DesyncV1Button

-- Fun√ß√£o Desync v1 (FUNCIONANDO)
local function toggleDesyncV1()
    desyncV1Active = not desyncV1Active
    
    if desyncV1Active then
        DesyncV1Button.Text = "üåÄ Desync v1: ON"
        DesyncV1Indicator.BackgroundColor3 = Color3.fromRGB(50, 180, 50)
        
        if player.Character then
            -- Salvar posi√ß√£o atual
            local char = player.Character
            local rootPart = char:FindFirstChild("HumanoidRootPart")
            
            if rootPart then
                -- Criar corpo falso parado
                fakeBody = Instance.new("Part")
                fakeBody.Name = "FakeBody"
                fakeBody.Size = Vector3.new(2, 4, 1)
                fakeBody.Position = rootPart.Position
                fakeBody.Transparency = 0.5
                fakeBody.Color = Color3.fromRGB(150, 150, 150)
                fakeBody.Material = EnumMaterial.Glass
                fakeBody.Anchored = true
                fakeBody.CanCollide = false
                fakeBody.Parent = workspace
                
                -- Tornar personagem invis√≠vel
                for _, part in pairs(char:GetChildren()) do
                    if part:IsA("BasePart") then
                        part.Transparency = 1
                        part.CanCollide = false
                    end
                end
                
                -- Atualizar posi√ß√£o do corpo falso
                spawn(function()
                    while desyncV1Active and fakeBody do
                        fakeBody.Position = rootPart.Position
                        task.wait(0.1)
                    end
                end)
            end
        end
    else
        DesyncV1Button.Text = "üåÄ Desync v1: OFF"
        DesyncV1Indicator.BackgroundColor3 = Color3.fromRGB(180, 50, 50)
        
        -- Remover corpo falso
        if fakeBody then
            fakeBody:Destroy()
            fakeBody = nil
        end
        
        -- Restaurar visibilidade
        if player.Character then
            for _, part in pairs(player.Character:GetChildren()) do
                if part:IsA("BasePart") then
                    part.Transparency = 0
                    part.CanCollide = true
                end
            end
        end
    end
end
